Why Python in Bioinformatics?

Python is the glue language of bioinformatics. While you'll use specialized tools (samtools, BWA, GATK) for heavy lifting, Python ties everything together:

  • Parsing — Read and write FASTA, FASTQ, GFF, VCF, and custom formats
  • Automation — Batch process hundreds of files, generate reports
  • API access — Query NCBI, Ensembl, UniProt programmatically
  • Custom analysis — When no existing tool does exactly what you need
  • Pysam, pybedtools — Python wrappers for samtools and BEDTools

Parsing FASTA Files

Manual Parsing (understand the format)

def parse_fasta(filepath):
    """Parse a FASTA file into a dict of {id: sequence}."""
    sequences = {}
    current_id = None
    with open(filepath) as f:
        for line in f:
            line = line.strip()
            if line.startswith('>'):
                current_id = line[1:].split()[0]  # ID = first word after >
                sequences[current_id] = ''
            elif current_id:
                sequences[current_id] += line
    return sequences

seqs = parse_fasta('sequences.fasta')
for name, seq in seqs.items():
    print(f"{name}: {len(seq)} bp, GC={100*(seq.count('G')+seq.count('C'))/len(seq):.1f}%")

Biopython SeqIO (the right way)

from Bio import SeqIO

# Parse FASTA — returns SeqRecord objects
for record in SeqIO.parse("sequences.fasta", "fasta"):
    seq = str(record.seq)
    gc = (seq.count('G') + seq.count('C')) / len(seq) * 100
    print(f"{record.id}: {len(seq)} bp, GC={gc:.1f}%")
    print(f"  Description: {record.description}")

# Load all sequences into a list
records = list(SeqIO.parse("sequences.fasta", "fasta"))
print(f"Total sequences: {len(records)}")

# Convert to dictionary for random access
seq_dict = SeqIO.to_dict(SeqIO.parse("sequences.fasta", "fasta"))
print(seq_dict["seq1"].seq[:50])  # First 50 bases of seq1

Parsing FASTQ Files

from Bio import SeqIO

# Parse FASTQ — includes quality scores
for record in SeqIO.parse("reads.fastq", "fastq"):
    quals = record.letter_annotations["phred_quality"]
    avg_qual = sum(quals) / len(quals)
    print(f"{record.id}: len={len(record.seq)}, avg_Q={avg_qual:.1f}")

# Filter reads by average quality
def filter_reads(input_fq, output_fq, min_qual=30):
    good = (r for r in SeqIO.parse(input_fq, "fastq")
            if sum(r.letter_annotations["phred_quality"]) / len(r) >= min_qual)
    count = SeqIO.write(good, output_fq, "fastq")
    print(f"Kept {count} reads with avg quality >= Q{min_qual}")

filter_reads("reads.fastq", "filtered.fastq", min_qual=30)

# Summary statistics
lengths = []
qualities = []
for record in SeqIO.parse("reads.fastq", "fastq"):
    lengths.append(len(record.seq))
    qualities.append(sum(record.letter_annotations["phred_quality"]) / len(record))

print(f"Reads: {len(lengths)}")
print(f"Length: min={min(lengths)}, max={max(lengths)}, mean={sum(lengths)/len(lengths):.0f}")
print(f"Quality: min={min(qualities):.1f}, max={max(qualities):.1f}, mean={sum(qualities)/len(qualities):.1f}")

Working with Sequences

from Bio.Seq import Seq

dna = Seq("ATGGCCATTGTAATGGGCCGCTGAAAGGGTGCCCGATAG")

# Transformations
print(f"Original:    {dna}")
print(f"Complement:  {dna.complement()}")
print(f"Rev comp:    {dna.reverse_complement()}")
print(f"Transcribed: {dna.transcribe()}")        # DNA → RNA (T→U)
print(f"Translated:  {dna.translate()}")          # DNA → Protein
print(f"Translated:  {dna.translate(to_stop=True)}")  # Stop at first stop codon

# GC content
from Bio.SeqUtils import gc_fraction
print(f"GC content:  {gc_fraction(dna)*100:.1f}%")

# Find motifs
pos = str(dna).find("ATG")  # Find start codon
print(f"First ATG at position: {pos}")

BLAST from Python

# Run BLAST locally (requires BLAST+ installed)
import subprocess

# Make database
subprocess.run(["makeblastdb", "-in", "ref.fa", "-dbtype", "nucl", "-out", "ref_db"])

# Run BLAST
subprocess.run([
    "blastn", "-query", "query.fa", "-db", "ref_db",
    "-outfmt", "6", "-evalue", "1e-5", "-out", "blast_results.txt"
])

# Parse tabular BLAST output
with open("blast_results.txt") as f:
    for line in f:
        fields = line.strip().split("\t")
        query, subject, pident, length, evalue = fields[0], fields[1], fields[2], fields[3], fields[10]
        print(f"{query} → {subject}: {pident}% identity, length={length}, E={evalue}")

# BLAST via NCBI web (slower, no local install needed)
from Bio.Blast import NCBIWWW, NCBIXML

result = NCBIWWW.qblast("blastn", "nt", "ATGCGATCGATCGATCG")
for record in NCBIXML.parse(result):
    for alignment in record.alignments[:5]:
        print(f"Hit: {alignment.title[:80]}")
        hsp = alignment.hsps[0]
        print(f"  Score={hsp.score}, E-value={hsp.expect}, Identity={hsp.identities}/{hsp.align_length}")

Automating File Processing

from pathlib import Path
from Bio import SeqIO
import subprocess

data_dir = Path("data/raw")
results_dir = Path("results/qc")
results_dir.mkdir(parents=True, exist_ok=True)

# Process all FASTQ files
for fq in sorted(data_dir.glob("*.fastq")):
    print(f"Processing {fq.name}...")
    
    # Run FastQC
    subprocess.run(["fastqc", str(fq), "-o", str(results_dir)], check=True)
    
    # Generate summary stats
    count, total_len, total_qual = 0, 0, 0
    for rec in SeqIO.parse(str(fq), "fastq"):
        count += 1
        total_len += len(rec.seq)
        total_qual += sum(rec.letter_annotations["phred_quality"]) / len(rec)
    
    avg_len = total_len / count if count else 0
    avg_qual = total_qual / count if count else 0
    print(f"  Reads: {count}, Avg length: {avg_len:.0f}, Avg quality: {avg_qual:.1f}")

Working with Pysam (SAM/BAM from Python)

import pysam

# Open a BAM file
bam = pysam.AlignmentFile("sorted.bam", "rb")

# Count mapped vs unmapped
mapped = 0
unmapped = 0
for read in bam:
    if read.is_unmapped:
        unmapped += 1
    else:
        mapped += 1
print(f"Mapped: {mapped}, Unmapped: {unmapped}")

# Fetch reads from a specific region
for read in bam.fetch("chr1", 1000, 2000):
    print(f"{read.query_name}: pos={read.reference_start}, MAPQ={read.mapping_quality}")

# Calculate coverage at each position
coverage = bam.count_coverage("chr1", 1000, 2000)
# Returns 4 arrays (A, C, G, T counts at each position)

bam.close()
Python vs R in Bioinformatics

Use Python for: file parsing, automation, pipeline scripting, API queries, custom tools, pysam/pybedtools
Use R for: statistics, visualization (ggplot2), Bioconductor packages (DESeq2, GenomicRanges), exploratory data analysis
Most bioinformaticians use both daily.
