Why R?

R is the language of statistical genomics. Bioconductor — R's bioinformatics ecosystem — has over 2,200 packages for genomic analysis. If you're doing differential expression, statistical testing, or publication-quality visualization, you're using R.

Data Wrangling with dplyr & tidyr

The tidyverse is a collection of R packages designed for data science. dplyr provides intuitive "verbs" for data manipulation:

library(dplyr)
library(tidyr)

# The pipe operator %>% chains operations (read as "then")
# filter: keep rows matching a condition
results %>% filter(padj < 0.05, abs(log2FoldChange) > 1)

# select: keep specific columns
results %>% select(gene, log2FoldChange, padj)

# mutate: create new columns
results %>% mutate(
  direction = ifelse(log2FoldChange > 0, "up", "down"),
  neg_log_p = -log10(padj)
)

# arrange: sort rows
results %>% arrange(padj)                    # Ascending (most significant first)
results %>% arrange(desc(abs(log2FoldChange))) # Largest fold changes first

# group_by + summarise: aggregate
counts_long %>%
  group_by(gene, condition) %>%
  summarise(
    mean_count = mean(count),
    sd_count = sd(count),
    n = n()
  )

# Combine operations
sig_genes <- results %>%
  filter(padj < 0.05, abs(log2FoldChange) > 1) %>%
  arrange(padj) %>%
  select(gene, log2FoldChange, padj) %>%
  mutate(direction = ifelse(log2FoldChange > 0, "up", "down"))

cat(nrow(sig_genes), "significant DE genes\n")

Visualization with ggplot2

ggplot2 implements the "Grammar of Graphics" — you build plots layer by layer:

library(ggplot2)

# Volcano plot — THE classic DE visualization
ggplot(results, aes(x=log2FoldChange, y=-log10(padj))) +
  geom_point(aes(color = padj < 0.05 & abs(log2FoldChange) > 1), 
             size=0.8, alpha=0.6) +
  scale_color_manual(values=c("grey70","red3"), 
                     labels=c("NS","Significant")) +
  geom_hline(yintercept=-log10(0.05), linetype="dashed", color="grey50") +
  geom_vline(xintercept=c(-1, 1), linetype="dashed", color="grey50") +
  labs(title="Differential Expression", 
       x="Log2 Fold Change", y="-Log10 Adjusted P-value",
       color="") +
  theme_minimal(base_size=12)

# Boxplot of gene expression by condition
ggplot(expr_data, aes(x=condition, y=log2(count+1), fill=condition)) +
  geom_boxplot(outlier.size=0.5) +
  facet_wrap(~gene, scales="free_y", ncol=3) +
  theme_minimal() +
  labs(y="Log2(count + 1)")

# MA plot (log2FC vs mean expression)
ggplot(results, aes(x=log10(baseMean), y=log2FoldChange)) +
  geom_point(aes(color=padj < 0.05), size=0.5, alpha=0.5) +
  geom_hline(yintercept=0, color="red") +
  theme_minimal()

# PCA plot (sample clustering)

Bioconductor Essentials

DESeq2 — Differential Expression

library(DESeq2)

# Create DESeq2 dataset
dds <- DESeqDataSetFromMatrix(
  countData = counts,        # Raw integer counts (NOT normalized)
  colData = metadata,        # Sample info (condition, batch, etc.)
  design = ~ condition       # Formula: what to test
)

# Pre-filter low-count genes (improves speed and multiple testing)
keep <- rowSums(counts(dds)) >= 10
dds <- dds[keep, ]

# Run DESeq2 pipeline (estimation, testing)
dds <- DESeq(dds)

# Extract results
res <- results(dds, contrast=c("condition", "treatment", "control"))
res <- res[order(res$padj), ]  # Sort by significance

# Summary
summary(res)

# Significant genes
sig <- subset(res, padj < 0.05 & abs(log2FoldChange) > 1)
cat(nrow(sig), "significant DE genes\n")
cat(sum(sig$log2FoldChange > 0), "upregulated\n")
cat(sum(sig$log2FoldChange < 0), "downregulated\n")

# Normalized counts for visualization
vsd <- vst(dds)  # Variance-stabilizing transformation
plotPCA(vsd, intgroup="condition")
Common DESeq2 Mistakes
  • Using normalized counts as input — DESeq2 needs RAW integer counts
  • Using pvalue instead of padj — always use the adjusted p-value (FDR)
  • Not filtering low-count genes — hurts multiple testing correction
  • Forgetting biological replicates — you need ≥3 per condition
